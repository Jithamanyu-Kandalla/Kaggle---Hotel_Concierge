Deployment Scripts for docker:

Dockerfile for backend service (backend/Dockerfile):
FROM python:3.9-slim

WORKDIR /app

COPY backend.py requirements.txt ./
RUN pip install -r requirements.txt

EXPOSE 6000

CMD ["python", "backend.py"]

Dockerfile for fulfillment service (fulfillment/Dockerfile):
FROM python:3.9-slim

WORKDIR /app

COPY fulfillment.py requirements.txt ./
RUN pip install -r requirements.txt

ENV GOOGLE_APPLICATION_CREDENTIALS=/app/credentials.json

EXPOSE 5000

CMD ["python", "fulfillment.py"]
